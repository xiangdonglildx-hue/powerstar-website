    ```javascript
    function renderTrendChart(dailyTrend) {
        if (!dailyTrend || dailyTrend.length === 0) {
            console.warn('No trend data available');
            return;
        }

        const ctx = document.getElementById('trendChart').getContext('2d');

        // Prepare data
        const labels = dailyTrend.map(d => {
            const date = new Date(d.date);
            return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        });
        const clicksData = dailyTrend.map(d => d.clicks);
        const impressionsData = dailyTrend.map(d => d.impressions);
        const positionData = dailyTrend.map(d => d.position);

        new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Clicks',
                        data: clicksData,
                        borderColor: '#ff4d00',  // orange
                        backgroundColor: 'rgba(255, 77, 0, 0.1)',
                        yAxisID: 'y',
                        tension: 0.3,
                        fill: false,
                    },
                    {
                        label: 'Impressions',
                        data: impressionsData,
                        borderColor: '#00fff0',  // cyan
                        backgroundColor: 'rgba(0, 255, 240, 0.1)',
                        yAxisID: 'y',
                        tension: 0.3,
                        fill: false,
                    },
                    {
                        label: 'Position',
                        data: positionData,
                        borderColor: '#888888',  // gray
                        backgroundColor: 'rgba(136, 136, 136, 0.1)',
                        yAxisID: 'y1',  // right axis
                        tension: 0.3,
                        fill: false,
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: {
                    mode: 'index',
                    intersect: false,
                },
                plugins: {
                    legend: {
                        labels: {
                            color: '#ffffff',
                            font: { family: 'Work Sans' }
                        }
                    }
                },
                scales: {
                    x: {
                        ticks: { color: '#888888' },
                        grid: { color: 'rgba(255, 255, 255, 0.05)' }
                    },
                    y: {
                        type: 'linear',
                        position: 'left',
    ```

    ```javascript
    function updateIndexingStatus(gsc) {
        if (!gsc) return;

        const indexed = gsc.indexedPages || 0;
        const total = gsc.totalSubmitted || 0;
        const unindexed = gsc.unindexedUrls || [];

        // Update summary
        document.getElementById('indexingSummary').textContent =
            `${indexed} of ${total} indexed`;

        // Update table
        const tbody = document.getElementById('unindexedTableBody');

        if (unindexed.length === 0) {
            tbody.innerHTML = '<tr><td colspan="2" class="no-data">All pages indexed</td></tr>';
            return;
        }

        tbody.innerHTML = unindexed.map(url => `
            <tr>
                <td class="url-cell">${escapeHtml(url)}</td>
                <td><span class="status-badge not-indexed">Not indexed</span></td>
            </tr>
        `).join('');
    }
    ```
